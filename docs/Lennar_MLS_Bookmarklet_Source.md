## How to Use This File

**This is the source of truth for all Matrix bookmarklet JavaScript.** It is never minified. When a launcher HTML file is needed for installation, extract the relevant function, minify it, and wrap it in the HTML launcher template.

**Two variants per tab:**
- **Non-Lennar** — full clipboard-reader; all fields pulled from payload; no hardcoded constants. Use for any non-Lennar listing.
- **Lennar** — same structure; Lennar constants hardcoded and fire automatically; only listing-specific dynamic fields come from clipboard. Use for all Lennar listings.

**Three entry paths — three bookmark folders per tab:**
- `Matrix — New` — clean slate; bookmarklet fills all fields
- `Matrix — Tax ID` — location fields pre-populated from tax record; bookmarklet skips pre-filled fields (except Owner Name, which always force-overwrites to `Lennar`)
- `Matrix — Copy` — most fields pre-populated from an existing listing; bookmarklet fills only listing-specific fields (price, address, sq ft, bath, remarks)

**The clipboard payload** is a single JSON object generated by the Claude session at listing intake. Each bookmarklet reads only its own top-level key. Copy the full payload once at the start of a listing session and use it across all tabs without re-copying.

**Map fields are always manual** — Latitude, Longitude, and Directions are never written by any bookmarklet. Human judgment required.

**Status tab is never automated** — MLS activation is Andrew's manual action only. Hard rule.

## Helper: setField(id, value)

Both variants use the same field-setting helpers. Define once at the top of each bookmarklet.

```javascript
// Set a text or select field value
function setField(id, value) {
  var el = document.getElementById(id);
  if (el) { el.value = value; }
}

// Set a checkbox state (check or uncheck)
function setCheck(id, checked) {
  var el = document.getElementById(id);
  if (el) { el.checked = checked; }
}

// Fire a change event on a select — required to trigger Matrix cascade dropdowns
function fireChange(id) {
  var el = document.getElementById(id);
  if (el) { el.dispatchEvent(new Event('change', { bubbles: true })); }
}

// Await a delay (used after firing cascade change events to allow DOM repopulation)
function wait(ms) {
  return new Promise(function(resolve) { setTimeout(resolve, ms); });
}
```

### Listing Info — Non-Lennar Variant

Reads all fields from clipboard payload. No hardcoded constants. Handles all three entry paths via a `path` key in the payload (`"new"`, `"taxid"`, `"copy"`).

```javascript
(function() {

  // --- Helpers ---
  function setField(id, value) {
    var el = document.getElementById(id);
    if (el) { el.value = value; }
  }
  function setCheck(id, checked) {
    var el = document.getElementById(id);
    if (el) { el.checked = checked; }
  }
  function fireChange(id) {
    var el = document.getElementById(id);
    if (el) { el.dispatchEvent(new Event('change', { bubbles: true })); }
  }
  function wait(ms) {
    return new Promise(function(resolve) { setTimeout(resolve, ms); });
  }

  navigator.clipboard.readText().then(function(text) {

    var payload = JSON.parse(text);
    var d = payload.listing;
    var path = payload.path || "new"; // "new", "taxid", or "copy"

    // --- Async cascade execution ---
    (async function() {

      // Step 1: County/City — must fire first; triggers Area, ZIP, Subdivision, Schools
      setField('Input_29', d.county_city);
      fireChange('Input_29');
      await wait(1500); // allow DOM to repopulate

      // Step 2: Area — triggers ZIP cascade on some configurations
      setField('Input_30', d.area);
      fireChange('Input_30');
      await wait(800);

      // Step 3: ZIP, Subdivision, Schools — set after cascade settles
      setField('Input_635', d.zip);
      setField('Input_259', d.subdivision);
      setField('Input_236', d.neighborhood || "");
      setField('Input_51',  d.elementary);
      setField('Input_53',  d.middle);
      setField('Input_52',  d.high);
      setField('Input_41',  d.post_office);

      // Step 4: Remaining Listing Information fields
      setField('Input_31',  d.list_price);
      setField('Input_160', d.list_date);
      setField('Input_849', d.type);
      setField('Input_850', d.attached_yn);
      setField('Input_162', d.expire_date);

      // Delayed Show — from payload (non-Lennar may vary)
      setField('Input_32', d.delayed_show || "0");

      // New/Resale — from payload
      setField('Input_42', d.new_resale || "");

      // Year Built Description — from payload (Lennar variant hardcodes UNDCON)
      setField('Input_45', d.year_built_desc || "");

      // PID / Tax ID
      if (path !== "taxid") {
        setField('Input_99', d.pid || "");
      }
      // On taxid path: Input_99 is pre-populated — skip

      // Step 5: Location fields
      // On taxid and copy paths, location fields are pre-populated — skip
      if (path === "new") {
        setField('Input_34', d.street_num);
        setField('Input_35', d.street_dir || "");
        setField('Input_36', d.street_name);
        setField('Input_37', d.street_suffix);
      }

      // Step 6: Property detail fields — always write regardless of path
      setField('Input_44', d.year_built);
      setField('Input_48', d.rooms);
      setField('Input_49', d.levels);
      setField('Input_622', d.lot);
      setField('Input_47', d.bedrooms);

      // SqFt Source — from payload
      setField('Input_97', d.sqft_source || "");

      // Square footage
      setField('Input_879', d.sqft_above_finished   || "0");
      setField('Input_882', d.sqft_below_finished   || "0");
      setField('Input_880', d.sqft_above_unfinished || "0");
      setField('Input_883', d.sqft_below_unfinished || "0");

    })();

  }).catch(function(e) {
    alert('Bookmarklet error — could not read clipboard: ' + e.message);
  });

})();
```
